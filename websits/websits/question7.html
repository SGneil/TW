題目 程式碼

請寫一程式，此程式從 b.txt 中讀取一個正的十進位數字(含小數)後將此十進位數字轉換成二進位數字，轉換後的數字請取至小數點第三位(小數點第三位以後的數字直接捨去)。
輸出格式為：轉換後的整數+轉換後的小數部分

範例輸入：
21.33
說明：
  21.33 轉換成二進位數字為 10101.010101…
  取小數點後三位後的數字為 10101.010
  因此輸出結果為 10101+010
範例輸出：
21.33
Ans=10101+010

(0)
範例輸入：
123.111
範例輸出：
123.111
Ans=1111011+000

(1)
範例輸入：
12.42
範例輸出：
12.42
Ans=1100+011

(2)
範例輸入：
123.111
範例輸出：
123.111
Ans=1111011+000

 
#include <stdio.h>
#include <string.h>

void print_num(char *str) {
    // 把字串轉成整數存到num
    int num = 0, ans[31], len = -1;
    for (int i = 0; i < strlen(str); i++) {
        num = num * 10 + (str[i] - '0');
    }

    // 把num轉成二進位存到ans(小數點前)
    while (num != 0) {
        ans[++len] = num % 2;
        num /= 2;
    }

    // 印出答案
    printf("Ans=");
    for (int i = len; i >= 0; i--) {
        printf("%d", ans[i]);
    }
    printf("+");
}

void print_float(char *str) {
    // 把字串轉成小數存到num
    int ans[3];
    double num = 0, point = 1;
    for (int i = 0; i < strlen(str); i++) {
        point /= 10;
        num += point * (str[i] - '0');
    }

    // 把num轉成二進位存到ans(小數點後)
    for (int i = 0; i < 3; i++) {
        num *= 2;
        if (num >= 1) {
            num -= 1;
            ans[i] = 1;
        }
        else {
            ans[i] = 0;
        }
    }

    // 印出答案
    for (int i = 0; i < 3; i++) {
        printf("%d", ans[i]);
    }
    printf("\n");
}

int main(void) {
    char str[201], *str_tok;
    gets(str);                  // 讀入輸入的資料
    printf("%s\n", str);

    // 把字串分割成小數點前跟小數點後
    str_tok = strtok(str, "."); // 小數點前
    print_num(str_tok);
    str_tok = strtok(NULL, ".");// 小數點後
    print_float(str_tok);
    return 0;
}
