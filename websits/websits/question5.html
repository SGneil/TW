題目 程式碼

試寫一程式，從檔案 e.txt 讀入一串整數，計算出此串整中某部分連續整數和為最大值，並由螢幕輸出。

範例輸入：
1 2 -6 3 -2 4 -1 3 2 -4
計算方法：最大值是 3+(-2)+4+(-1)+3+2=9
範例輸出：
1 2 -6 3 -2 4 -1 3 2 -4
Ans=9

(0)
範例輸入：
1 2 -6 3 -2 4 -1 3 2 -4
範例輸出：
1 2 -6 3 -2 4 -1 3 2 -4
Ans=9

(1)
範例輸入：
1 -7 9 3 -2 9 -1 -7 -2 -4
範例輸出：
1 -7 9 3 -2 9 -1 -7 -2 -4
Ans=19

(2)
範例輸入：
-5 -7 9 8 -2 9 -1 -7 9 -3
範例輸出：
-5 -7 9 8 -2 9 -1 -7 9 -3
Ans=25

 
#include <stdio.h>

int main(void) {
    int num[201], len = -1, sum = 0;
    while (scanf("%d", &num[++len]) != EOF);

    // 印出輸入資料
    for (int i = 0; i < len; i++) {
        printf("%d", num[i]);
        if (i != len - 1) {
            printf(" ");
        }
    }
    printf("\n");

    // 計算所有的連續整數和，並把最大值存到sum變數中
    for (int i = 0; i < len; i++) {
        for (int j = i, temp = 0; j < len; j++) {
            temp += num[j];
            if (temp > sum) {
                sum = temp;
            }
        }
    }

    printf("Ans=%d\n", sum);
    return 0;
}
