題目 程式碼

小明有一天拿到一份資料，是由 0 跟 1 組成的二維陣列，表示一個地圖上面有道路的地方，0 表示沒有馬路，1 表示有馬路，接著他想要找出最長的一條路，這條路只能是水平線，垂直線或者對角線，請撰寫一個程式來協助小明解決問題。
請從 e.txt 中先讀入寬與高(皆不會超過 30)，接著讀入這個二維陣列，然後輸出最長的一條路。

範例輸入：
6 14
01001110001000
00100100010000
01110110111000
00001000000100
00000100001010
00111110010000
範例輸出：
6 14
01001110001000
00100100010000
01110110111000
00001000000100
00000100001010
00111110010000

#include <stdio.h>

int main(void) {
    // 讀入輸入資料
    int num1, num2, max = 0, ans = 0;
    scanf("%d %d", &num1, &num2);
    int arr[num1][num2];
    for (int i = 0; i < num1; i++) {
        for (int j = 0; j < num2; j++) {
            scanf("%1d", &arr[i][j]);
        }
    }

    // 找出水平線的最大值
    for (int i = 0; i < num1; i++) {
        int temp = 0;
        for (int j = 0; j < num2; j++) {
            if (arr[i][j] == 0) {
                if (max < temp) {
                    max = temp;
                }
                temp = 0;
            }
            else {
                temp++;
            }
        }
        if (temp > max) {
            max = temp;
        }
    }
    if (ans < max) {
        ans = max;
    }

    // 找出垂直線的最大值
    max = 0;
    for (int i = 0; i < num2; i++) {
        int temp = 0;
        for (int j = 0; j < num1; j++) {
            if (arr[j][i] == 0) {
                if (max < temp) {
                    max = temp;
                }
                temp = 0;
            }
            else {
                temp++;
            }
        }
        if (temp > max) {
            max = temp;
        }
    }
    if (ans < max) {
        ans = max;
    }

    // 找出對角線的最大值(往右斜)
    max = 0;
    for (int i = 0; i <= num2 - num1; i++) {
        int temp = 0;
        for (int j = 0; j < num1; j++) {
            if (arr[j][i + j] == 0) {
                if (temp > max) {
                    max = temp;
                }
                temp = 0;
            }
            else {
                temp++;
            }
        }
        if (temp > max) {
            max = temp;
        }
    }
    if (ans < max) {
        ans = max;
    }

    // 找出對角線的最大值(往左斜)
    max = 0;
    for (int i = num2 - 1; i >= num1 - 1; i--) {
        int temp = 0;
        for (int j = 0; j < num1; j++) {
            if (arr[j][i - j] == 0) {
                if (temp > max) {
                    max = temp;
                }
                temp = 0;
            }
            else {
                    temp++;
                }
        }
        if (temp > max) {
            max = temp;
        }
    }
    if (ans < max) {
        ans = max;
    }

    // 印出答案
    printf("%d %d\n", num1, num2);
    for (int i = 0; i < num1; i++) {
        for (int j = 0; j < num2; j++) {
            printf("%d", arr[i][j]);
        }
        printf("\n");
    }
    printf("Ans=%d\n", ans);
    return 0;
}
