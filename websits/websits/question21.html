題目 程式碼

猜數字：請撰寫一個程式從 a.txt 中讀入兩個數字字串（長度最長為 10 且數字不重複），並且判斷字串中的每個數字，若是兩個字串在同一個位置的數字都相同，則記載一個 A，若是兩個字串都有這個數字但是位置不同，則記載一個 B，當沒有 A 也沒有 B 時則輸出0A0B。

範例輸入：
14685
47653
範例輸出：
14685
47653
Ans=1A2B

(0)
範例輸入：
19234
23574
範例輸出：
19234
23574
Ans=1A2B

(1)
範例輸入：
85672431
36452891
範例輸出：
85672431
36452891
Ans=2A5B

(2)
範例輸入：
75294
73986
範例輸出：
75294
73986
Ans=1A1B

 
#include <stdio.h>
#include <string.h>
// 計算A的數量
int count_A(char *str1, char *str2) {
    int sum = 0;
    for (int i = 0; i < strlen(str1); i++) {
        if (str1[i] == str2[i]) {
            sum++;
        }
    }
    return sum;
}

// 計算B的數量
int count_B(char *str1, char *str2) {
    int sum = 0;
    for (int i = 0; i < strlen(str1); i++) {
        for (int j = 0; j < strlen(str1); j++) {
            if (str1[i] == str2[j] && i != j) {
                sum++;
                break;
            }
        }
    }
    return sum;
}

int main(void) {
    // 讀入輸入資料
    char str1[11], str2[11];
    scanf("%s %s", str1, str2);

    // 印出答案
    printf("%s\n", str1);
    printf("%s\n", str2);
    printf("Ans=%dA%dB\n", count_A(str1, str2), count_B(str1, str2));
    return 0;
}
