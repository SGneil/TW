題目 程式碼

恐怖份子在一條巷子中放入了好幾顆炸彈，炸彈有分小炸彈與大炸彈，請計算這條巷子在所有炸彈引爆後，還有那些地方是安全的。這裡假設巷子以一連串的 0 來表示，小 x表示小炸彈，爆炸範圍是 1(會炸到左右各一個 0)，大 X 是大炸彈，爆炸範圍是 2(會炸到左右各 2 個 0)，請從 d.txt 中讀入一串由 0, x, X 組成的字串(最長 80 個字元)，接著輸出炸彈爆炸後還剩下幾個 0。

範例輸入一：
00x00000X0xX0000x

上面的炸彈爆炸後的結果為:
0---00--------0--
剩下 4 個 0

範例輸出一：
00x00000X0xX0000x
Ans=4

(0)
範例輸入：
00x00000X0xX0000x
範例輸出：
00x00000X0xX0000x
Ans=4

(1)
範例輸入：
0x0
範例輸出：
0x0
Ans=0

(2)
範例輸入：
000000000000000000000000000000
範例輸出：
000000000000000000000000000000
Ans=30

 
#include <stdio.h>
#include <string.h>

void meet_x(char * str, int index) {    // 遇到小炸彈
    str[index] = '-';
    if (index > 0) {    // 確認炸掉的範圍有沒有超出索引值
        str[index - 1] = '-';
    }
    if (index < strlen(str) - 1) {    // 確認炸掉的範圍有沒有超出索引值
        if (str[index + 1] != 'x' && str[index + 1] != 'X') {   // 不要炸掉之後的炸彈
            str[index + 1] = '-';
        }
    }
}

void meet_X(char * str, int index) {    // 遇到大炸彈
    str[index] = '-';
    if (index > 1) {    // 確認炸掉的範圍有沒有超出索引值
        str[index - 1] = '-';
        str[index - 2] = '-';
    }
    else if (index > 0) {    // 確認炸掉的範圍有沒有超出索引值
        str[index - 1] = '-';
    }

    if (index < strlen(str) - 2) {    // 確認炸掉的範圍有沒有超出索引值
        if (str[index + 1] != 'X') {
            str[index + 1] = '-';
        }
        if (str[index + 2] != 'x' && str[index + 2] != 'X') {   // 不要炸掉炸彈之後的炸彈
            str[index + 2] = '-';
        }
    }
    else if (index < strlen(str) - 1) { // 確認炸掉的範圍有沒有超出索引值
        if (str[index + 1] != 'X') {
            str[index + 1] = '-';
        }
    }
}

int main(void) {
    char str[81];       // 讀入輸入資料
    gets(str);
    printf("%s\n", str);

    for (int i = 0; i < strlen(str); i++) {
        if (str[i] == 'x') {
            meet_x(str, i);
        }
        else if (str[i] == 'X') {
            meet_X(str, i);
        }
    }

    // 計算有幾個0
    int ans = 0;
    for (int i = 0; i < strlen(str); i++) {
        if (str[i] == '0') {
            ans++;
        }
    }
    printf("Ans=%d\n", ans);
    return 0;
}
