題目 程式碼

部分序列字串指原字串中刪掉一個以上字元且不改變原順序的部份字串。請撰寫程式從資料檔 d.txt 讀入兩個字串(每一字串皆不超過 50 字元)，判斷第一個字串是否為第二個字串的部份序列字串，若是則輸出 Yes 以及被刪除之部分序列字串，否則輸出 No。

範例輸入一：
bcfg gcbgcefgjvk
範例輸出一：
bcfg gcbgcefgjvk
Ans=Yes gcgejvk
範例輸入二：
bcfg gcagcefgjvk
範例輸出二：
bcfg gcagcefgjvk
Ans=No

bcfg gcagcefgjvk

bcfg gcagcefgjvk

abcd hiajbcloide

Ans=Yes lreoeioedeeqzcbmnoecn

 
#include <stdio.h>
#include <string.h>
// 判斷str1是否為str2的部分字串
int is_part_str(char *str1, char *str2) {
    for (int i = 0, j = 0; i < strlen(str2); i++) {
        if (str1[j] == str2[i]) {
            j++;
        }
        if (j == strlen(str1) - 1) {
            return 1;
        }
    }
    return 0;
}

// 印出刪除子字串後的字串
void ans_str(char *str1, char * str2) {
    for (int i = 0, j = 0; i < strlen(str2); i++) {
        if (str1[j] == str2[i] && j != strlen(str1)) {
            j++;
        }
        else {
            printf("%c", str2[i]);
        }
    }
    printf("\n");
}

int main(void) {
    // 讀入輸入資料
    char str1[51], str2[51];
    scanf("%s %s", str1, str2);

    // 印出第一行答案
    printf("%s %s\n", str1, str2);

    // 判斷是否為子字串，決定印出Yes或No
    if (is_part_str(str1, str2)) {
        printf("Ans=Yes ");
        ans_str(str1, str2);
    }
    else {
        printf("Ans=No\n");
    }
    return 0;
}
