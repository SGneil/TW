題目 程式碼

分別自檔案 e.txt 讀入一串由()[]{}所組成之字串，判斷字串是否合法，一個()[]{}必須對稱存在，且對應的括號間不能插入單獨的括號。

範例輸入一：
({[()]()})
範例輸出一：
({[()]()})
Ans=CORRECT

範例輸入二：
(){[()]()
範例輸出二：
(){[()]()
Ans=ERROR

(0)
範例輸入：
({[()]()})
範例輸出：
({[()]()})
Ans=CORRECT

(1)
範例輸入：
(){[()]()
範例輸出：
(){[()]()
Ans=ERROR

(2)
範例輸入：
()[]{}([{}])[(])
範例輸出：
()[]{}([{}])[(])
Ans=ERROR

(3)
範例輸入：
({[][](){}})
範例輸出：
({[][](){}})
Ans=CORRECT

(4)
範例輸入：
[{}{{([(]))}}]
範例輸出：
[{}{{([(]))}}]
Ans=ERROR

 
#include <stdio.h>
#include <string.h>
// 判斷字串是否合法
int check(char *str) {
    char str_temp[201];
    for (int i = 0, j = -1; i < strlen(str); i++) {
        // 把遇到的 ( [ { 存到str_temp
        if (str[i] == '(' || str[i] == '[' || str[i] == '{') {
            str_temp[++j] = str[i];
        }

        // 當遇到 ) ] } 比對上一個存到str_temp的括號，如果相減 <=2 代表符合
        else if (str[i] == ')' || str[i] == ']' || str[i] == '}') {
            if (str[i] - str_temp[j] <= 2) {
                j--;
            }
            else if (str[i] - str_temp[j] > 2) {
                return 0;
            }
        }

        // 比對到最後一個時 j == -1 代表每個括號都比對而且全部正確
        if (i == strlen(str) - 1 && j == -1) {
            return 1;
        }
    }
    return 0;
}

int main(void) {
    // 讀入輸入資料
    char str[201];
    gets(str);
    printf("%s\n", str);

    // 印出答案
    if (check(str)) {
        printf("Ans=CORRECT\n");
    }
    else {
        printf("Ans=ERROR\n");
    }
    return 0;
}
