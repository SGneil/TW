題目 程式碼

試寫一程式，從檔案(c.txt)讀入一串正整數(最多 10 個)，試找出此串正整數之最小公倍數。例如：

範例輸入：
12 16 36 128
範例輸出：
12 16 36 128
Ans=1152

(0)
範例輸入：
12 16 36 128
範例輸出：
12 16 36 128
Ans=1152

(1)
範例輸入：
153 13 413 646
範例輸出：
153 13 413 646
Ans=31215366

(2)
範例輸入：
78 95 13 84
範例輸出：
78 95 13 84
Ans=103740

 
#include <stdio.h>
// 確認max是否為最大公倍數
int check(int num[10], int max, int len) {
    for (int i = 0; i < len; i++) {
        if (max % num[i] != 0) {
            return 1;
        }
    }
    return 0;
}

int main(void) {
    // 讀入輸入資料
    int num[10], len = -1;
    while (scanf("%d", &num[++len]) != EOF);

    // 印出部分答案
    for (int i = 0; i < len; i++) {
        printf("%d", num[i]);
        if (i != len - 1) {
            printf(" ");
        }
        else {
            printf("\n");
        }
    }

    int max = num[0];
    for (int i = 1; i < len; i++) {
        if (max < num[i]) {
            max = num[i];
        }
    }

    // 找最大公倍數
    while (check(num, max, len)) {
        max++;
    }

    // 印出答案
    printf("Ans=%d\n", max);
    return 0;
}
